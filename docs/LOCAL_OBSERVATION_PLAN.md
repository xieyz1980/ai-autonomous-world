## 🔧 监控脚本

### 实时观察命令

```bash
# 1. 实时日志观察
tail -f /opt/ai-world/logs/kingdom2.log

# 2. 每5分钟自动检查
cat > /tmp/watchdog.sh << 'EOF'
#!/bin/bash
while true; do
    echo "[$(date)] 检查系统状态..."
    
    # 检查进程
    if pgrep -f "main_kingdom" > /dev/null; then
        echo "✓ 进程运行正常"
    else
        echo "✗ 进程异常！"
    fi
    
    # 检查日志更新
    LOG_LINES=$(wc -l < /opt/ai-world/logs/kingdom2.log)
    echo "✓ 日志行数: $LOG_LINES"
    
    # 检查任务生成
    TASK_COUNT=$(grep -c "Task submitted" /opt/ai-world/logs/kingdom2.log)
    echo "✓ 任务生成数: $TASK_COUNT"
    
    echo "---"
    sleep 300  # 5分钟
done
EOF

chmod +x /tmp/watchdog.sh
/tmp/watchdog.sh &
```

### 每日报告生成

```bash
# 每天生成观察报告
cat > /tmp/daily_report.sh << 'EOF'
#!/bin/bash
DATE=$(date +%Y%m%d)
REPORT="/opt/ai-world/reports/daily_${DATE}.md"

mkdir -p /opt/ai-world/reports

cat > $REPORT << REPORT
# AI Kingdom 观察日报 - $DATE

## 运行概况
- 报告时间: $(date)
- 日志文件: kingdom2.log
- 总行数: $(wc -l < /opt/ai-world/logs/kingdom2.log)

## 关键指标
- 任务生成数: $(grep -c "Task submitted" /opt/ai-world/logs/kingdom2.log)
- 任务分配数: $(grep -c "assigned to" /opt/ai-world/logs/kingdom2.log)
- 执行次数: $(grep -c "Executing task" /opt/ai-world/logs/kingdom2.log)
- 错误数: $(grep -c "ERROR" /opt/ai-world/logs/kingdom2.log || echo 0)

## 最后10条日志
\`\`\`
$(tail -10 /opt/ai-world/logs/kingdom2.log)
\`\`\`

## 观察结论
[待填写]

## 明日计划
[待填写]
REPORT

echo "日报已生成: $REPORT"
EOF

chmod +x /tmp/daily_report.sh

# 添加到定时任务 (每天23:59生成)
echo "59 23 * * * /tmp/daily_report.sh" | crontab -
```
